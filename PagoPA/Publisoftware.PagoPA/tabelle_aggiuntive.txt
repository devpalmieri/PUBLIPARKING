CREATE TABLE [dbo].[tab_pagopa_eventi]
(
	id_evento BIGINT NOT NULL IDENTITY(1, 1) PRIMARY KEY,
	id_carrello INT,
	componente VARCHAR(35),
	tipo_evento VARCHAR(70),
	data DATETIME,
	esito VARCHAR(4),
	dettaglio_esito TEXT,
	parametri_richiesta VARBINARY(max),
	parametri_risposta VARBINARY(max),
	dati_pago_pa TEXT,
	iuv VARCHAR(35),
	ccp VARCHAR(35),
	dominio VARCHAR(35)
	CONSTRAINT fk_carrello FOREIGN KEY (id_carrello) REFERENCES [dbo].[tab_carrello] (id_carrello)
)

CREATE TABLE [dbo].[anagrafica_pagopa_intermediari]
(
	id_intermediario INT NOT NULL IDENTITY(1, 1) PRIMARY KEY,
	cod_intermediario VARCHAR(35) NOT NULL,
	cod_connettore_pdd VARCHAR(35) NOT NULL,
	cod_connettore_ftp VARCHAR(35),
	denominazione VARCHAR(255) NOT NULL,
	id_ente INT NOT NULL,
	CONSTRAINT fk_ente FOREIGN KEY (id_ente) REFERENCES [dbo].[anagrafica_ente] (id_ente)
)


CREATE TABLE [dbo].[anagrafica_pagopa_stazioni]
(
	id_stazione INT NOT NULL IDENTITY(1, 1) PRIMARY KEY,
	cod_stazione VARCHAR(35) NOT NULL,
	[password] VARCHAR(35) NOT NULL,
	id_intermediario INT NOT NULL,
	CONSTRAINT fk_intermediario FOREIGN KEY (id_intermediario) REFERENCES [dbo].[anagrafica_pagopa_intermediari] (id_intermediario)
)
